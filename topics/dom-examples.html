10. Remove an Event Listener


<button id="example10">Click Me (Listener will be removed)</button>
<script>
    function clickHandler() {
        alert("Listener Active");
    }
    const button = document.getElementById("example10");
    button.addEventListener("click", clickHandler);
    setTimeout(() => {

    }, 5000);
</script>

11. Clone an Element

12. Replace an Element

<script>
    function replaceElement() {
        let newElement = document.createElement("div");
        newElement.textContent = "I replaced the old one!";

        let oldElement = document.getElementById("example12");
        container.replaceChild(newElement, oldElement);

</script>

</script>

20. Create and Append Table Rows


<table id="example20" border="1">
    <thead>
        <tr>
            <th>Name</th>
            <th>Age</th>
        </tr>
    </thead>
    <tbody>
    </tbody>
</table>
<button onclick="addTableRows()">Add Rows</button>
<script>
    function addTableRows() {
        let table = document.getElementById("example20").getElementsByTagName("tbody")[0];
        let data = [
            { name: "Alice", age: 25 },
            { name: "Bob", age: 30 }
        ];
        data.forEach(item => {
            let row = document.createElement("tr");
            let nameCell = document.createElement("td");
            nameCell.textContent = item.name;
            let ageCell = document.createElement("td");
            ageCell.textContent = item.age;
            row.appendChild(nameCell);
            row.appendChild(ageCell);
            table.appendChild(row);
        });
    }
</script>
